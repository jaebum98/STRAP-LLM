##### Convert to python function format #####

def put_egg_in_fridge(robot_name):
    GoToObject('Egg')
    PickupObject('Egg')
    GoToObject('Fridge')
    OpenObject('Fridge')
    PutObject('Egg', 'Fridge')
    CloseObject('Fridge')

def slice_apple(robot_name):
    GoToObject('Knife')
    PickupObject('Knife')
    GoToObject('Apple')
    SliceObject('Apple')
    GoToObject('DiningTable')
    PutObject('Knife', 'DiningTable')

def put_apple_in_pot(robot_name):
    GoToObject('Apple')
    PickupObject('Apple')
    GoToObject('Pot')
    PutObject('Apple', 'Pot')

def place_pot_in_fridge(robot_name):
    PickupObject('Pot')
    GoToObject('Fridge')
    OpenObject('Fridge')
    PutObject('Pot', 'Fridge')
    CloseObject('Fridge')

# Task Execution Sequence
def execute_task():
    put_egg_in_fridge(robot1)
    slice_apple(robot1)
    put_apple_in_pot(robot1)
    place_pot_in_fridge(robot8)
